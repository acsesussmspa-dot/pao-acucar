function doPost(e) { var lock = LockService.getScriptLock(); lock.waitLock(10000); try { var dados = JSON.parse(e.postData.contents); var planilha = SpreadsheetApp.openById("15brMN7WjaVGoaU2vgn8lvDyo92kza7QCwYAYt-q2IhA"); var aba = planilha.getSheetByName("TRATATIVAS"); // dados.ocorrencias é um array — pode vir 1 ou várias ocorrências no mesmo envio var ocorrencias = dados.ocorrencias || [dados]; // compatível com envio antigo (sem array) ocorrencias.forEach(function(ocorrencia) { // Formata a data (yyyy-mm-dd vinda do input) para dd/mm/yyyy var dataFormatada = ocorrencia.data; if (ocorrencia.data) { var partes = ocorrencia.data.split("-"); dataFormatada = partes[2] + "/" + partes[1] + "/" + partes[0]; } aba.appendRow([ dataFormatada, // DATA DA OCORRÊNCIA dados.unidade, // UNIDADE dados.nome, // NOME DO PROFISSIONAL dados.matricula, // MATRÍCULA ocorrencia.entrada1, // ENTRADA 1º ocorrencia.saida1, // SAÍDA 1º ocorrencia.entrada2, // ENTRADA 2º ocorrencia.saida2, // SAÍDA 2º ocorrencia.ocorrencia, // OCORRÊNCIA ocorrencia.justificativa, // JUTIFICATIVA dados.email, // EMAIL DE CONFIRMAÇÃO "" // ENVIAR P/ PLANILHA (deixa em branco) ]); }); // Envia e-mail de confirmação para quem preencheu o formulário if (dados.email) { var corpo = "Olá, " + dados.nome + "!\n\n"; corpo += "Recebemos sua(s) tratativa(s) de ponto na unidade " + dados.unidade + ".\n\n"; ocorrencias.forEach(function(ocorrencia, i) { corpo += "Ocorrência " + (i + 1) + ":\n"; corpo += "- Data: " + ocorrencia.data + "\n"; corpo += "- Tipo: " + ocorrencia.ocorrencia + "\n"; corpo += "- Justificativa: " + ocorrencia.justificativa + "\n\n"; }); corpo += "Qualquer dúvida, entre em contato com a coordenação.\n\nAtenciosamente,\nCoordenação Esf Pão de Açúcar"; MailApp.sendEmail({ to: dados.email, subject: "Confirmação de Tratativa de Ponto - Esf Pão de Açúcar", body: corpo }); } return ContentService .createTextOutput(JSON.stringify({ status: "ok" })) .setMimeType(ContentService.MimeType.JSON); } catch (erro) { return ContentService .createTextOutput(JSON.stringify({ status: "erro", mensagem: erro.toString() })) .setMimeType(ContentService.MimeType.JSON); } finally { lock.releaseLock(); } }
